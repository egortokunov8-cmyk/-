<script>
    // ---------- ТАРИФЫ (статичные) ----------
    const tariffs = [
        { name: "Уличный боец", price: "3900 ₽/мес", trainer: "Тайлер Дёрден", phone: "+1 (555) 724-1892", action: "🔥 Первый бой в подарок" },
        { name: "Мясник", price: "5900 ₽/мес", trainer: "Марла Сингер", phone: "+1 (555) 983-4512", action: "🧼 Скидка 20% на мыло" },
        { name: "Проект 'Майхем'", price: "8900 ₽/мес", trainer: "Роберт Полсон", phone: "+1 (555) 347-0085", action: "💣 Бесплатный бинт" }
    ];

    // 4 тренировки
    const workoutsCatalog = [
        { id: "w1", name: "Уличный бокс", desc: "Работа на мешках, защита, взрывная серия", price: "1100 ₽", icon: "fas fa-fist-raised" },
        { id: "w2", name: "ММА / Грэпплинг", desc: "Борьба, партер, болевые приёмы", price: "1400 ₽", icon: "fas fa-hand-fist" },
        { id: "w3", name: "Муай Тай", desc: "Колени, локти — искусство 8 конечностей", price: "1250 ₽", icon: "fas fa-user-ninja" },
        { id: "w4", name: "Кроссфит-уничтожение", desc: "Функционалка до отказа", price: "950 ₽", icon: "fas fa-bolt" }
    ];

    // ГЛОБАЛЬНАЯ БАЗА ДАННЫХ (массив клиентов)
    let clients = [];

    // Получить стандартных 5 клиентов (как в задании)
    function getDefaultFive() {
        return [
            { id: 1, name: "Тайлер Дёрден-мл.", isMember: true, purchasedWorkouts: ["Уличный бокс"] },
            { id: 2, name: "Майк 'Бритва'", isMember: true, purchasedWorkouts: ["ММА / Грэпплинг", "Муай Тай"] },
            { id: 3, name: "Андрей 'Стич'", isMember: false, purchasedWorkouts: [] },
            { id: 4, name: "Елена 'Кулак'", isMember: false, purchasedWorkouts: [] },
            { id: 5, name: "Виктор 'Угрюмый'", isMember: false, purchasedWorkouts: [] }
        ];
    }

    // Загрузка/сохранение (localStorage)
    function saveToLocal() {
        localStorage.setItem("fightClubDB_v2", JSON.stringify(clients));
    }

    function loadFromLocal() {
        const saved = localStorage.getItem("fightClubDB_v2");
        if (saved) {
            try {
                clients = JSON.parse(saved);
                clients.forEach(c => { if (!c.purchasedWorkouts) c.purchasedWorkouts = []; });
            } catch(e) { clients = getDefaultFive(); }
        } else {
            clients = getDefaultFive();
        }
        // Убедимся, что нет дублирующегося пользователя "Я" с id 999 (если был)
        // и проверим, существует ли профиль текущего пользователя? будем хранить отдельно в переменной
    }

    // Сохраняем ID текущего пользователя (если он создал профиль)
    let currentUserId = null;

    // Обновляем отображение твоего профиля
    function updateMyProfileUI() {
        const container = document.getElementById("myProfileInfo");
        if (currentUserId !== null) {
            const me = clients.find(c => c.id === currentUserId);
            if (me) {
                container.innerHTML = `
                    <div><i class="fas fa-user"></i> <strong>${me.name}</strong> 
                    ${me.isMember ? '<span class="badge-member"><i class="fas fa-check-circle"></i> Абонемент активен</span>' : '<span class="badge-nonmember"><i class="fas fa-hourglass"></i> Нет абонемента</span>'}
                    </div>
                    <div><i class="fas fa-ticket"></i> Твои тренировки: ${me.purchasedWorkouts.length ? me.purchasedWorkouts.join(", ") : "—"}</div>
                    ${!me.isMember ? '<button id="buyMembershipForMeBtn" class="btn-primary" style="margin-top:10px; width:auto; padding:6px 20px;">💪 Купить абонемент сейчас</button>' : ''}
                `;
                const buyBtn = document.getElementById("buyMembershipForMeBtn");
                if (buyBtn) buyBtn.onclick = () => buyMembershipForCurrentUser();
                return;
            }
        }
        // если профиля нет
        container.innerHTML = `⚠️ Введи имя и нажми «Стать членом клуба», чтобы купить абонемент и покупать тренировки под своим именем.`;
    }

    // Функция покупки абонемента для текущего пользователя
    function buyMembershipForCurrentUser() {
        if (currentUserId === null) {
            alert("Сначала создай профиль: введи имя и нажми 'Стать членом клуба'.");
            return;
        }
        const me = clients.find(c => c.id === currentUserId);
        if (me && !me.isMember) {
            me.isMember = true;
            saveToLocal();
            updateMyProfileUI();
            renderClientsList(); // обновить общую БД
            alert(`🥊 Поздравляю, ${me.name}! Теперь ты член Бойцовского клуба. Можешь покупать тренировки.`);
        } else if (me && me.isMember) {
            alert("У тебя уже есть абонемент, боец!");
        }
    }

    // Создать или обновить профиль "Я"
    function createOrUpdateMyProfile(userName) {
        if (!userName.trim()) {
            alert("Введи имя!");
            return;
        }
        // Проверим, есть ли уже профиль с таким именем? но нам нужен отдельный
        if (currentUserId !== null) {
            // обновим имя существующего
            const me = clients.find(c => c.id === currentUserId);
            if (me) {
                me.name = userName.trim();
                saveToLocal();
                updateMyProfileUI();
                renderClientsList();
                alert(`Имя обновлено на "${me.name}"`);
            } else {
                // ошибочный кейс: создадим заново
                createNewUserProfile(userName);
            }
        } else {
            createNewUserProfile(userName);
        }
    }

    function createNewUserProfile(userName) {
        // ищем максимальный id
        const maxId = clients.length > 0 ? Math.max(...clients.map(c => c.id)) : 5;
        const newId = maxId + 1;
        const newUser = {
            id: newId,
            name: userName.trim(),
            isMember: false,
            purchasedWorkouts: []
        };
        clients.push(newUser);
        currentUserId = newId;
        saveToLocal();
        updateMyProfileUI();
        renderClientsList();
        alert(`✅ Профиль "${userName}" создан! Теперь можешь купить абонемент (кнопка ниже) и тренировки.`);
    }

    // Покупка тренировки (для любого члена клуба)
    function openBuyWorkoutModal(workoutName) {
        const members = clients.filter(c => c.isMember === true);
        if (members.length === 0) {
            alert("Нет активных членов клуба. Сначала кто-то должен купить абонемент (например, ты).");
            return;
        }
        const select = document.getElementById("clientSelectModal");
        select.innerHTML = "";
        members.forEach(m => {
            const option = document.createElement("option");
            option.value = m.id;
            option.textContent = `${m.name} (тренировки: ${m.purchasedWorkouts.length ? m.purchasedWorkouts.join(", ") : "—"})`;
            select.appendChild(option);
        });
        document.getElementById("modalWorkoutName").innerText = `Тренировка: ${workoutName}`;
        window.selectedWorkout = workoutName;
        document.getElementById("purchaseModal").style.display = "flex";
    }

    function confirmPurchase() {
        const select = document.getElementById("clientSelectModal");
        const clientId = parseInt(select.value);
        const workout = window.selectedWorkout;
        if (!workout) return;
        const client = clients.find(c => c.id === clientId);
        if (client && client.isMember) {
            if (!client.purchasedWorkouts.includes(workout)) {
                client.purchasedWorkouts.push(workout);
                saveToLocal();
                renderClientsList();
                updateMyProfileUI();
                alert(`✅ ${client.name} приобрёл тренировку "${workout}"! Ударная мощь растёт.`);
            } else {
                alert(`❌ ${client.name} уже имеет эту тренировку.`);
            }
        } else {
            alert("Клиент без абонемента не может покупать тренировки.");
        }
        closeModal();
    }

    function closeModal() {
        document.getElementById("purchaseModal").style.display = "none";
        window.selectedWorkout = null;
    }

    // Рендер тарифов
    function renderTariffs() {
        const container = document.getElementById("tariffsContainer");
        container.innerHTML = tariffs.map(t => `
            <div class="tariff-card">
                <div class="tariff-name">${t.name}</div>
                <div class="trainer-info"><i class="fas fa-chalkboard-user"></i> Тренер: ${t.trainer}<br><small>${t.phone}</small></div>
                <div class="price">${t.price}</div>
                <div class="sale">🎯 ${t.action}</div>
            </div>
        `).join("");
    }

    // Рендер тренировок
    function renderWorkouts() {
        const container = document.getElementById("workoutsList");
        container.innerHTML = workoutsCatalog.map(w => `
            <div class="workout-card">
                <div class="workout-icon" style="font-size:2rem; text-align:center;"><i class="${w.icon}"></i></div>
                <div class="workout-info" style="padding:0.5rem;">
                    <h3>${w.name}</h3>
                    <p style="font-size:0.8rem;">${w.desc}</p>
                    <div class="price">${w.price}</div>
                    <button class="btn-buy-workout" data-workout="${w.name}">Купить тренировку</button>
                </div>
            </div>
        `).join("");
        document.querySelectorAll(".btn-buy-workout").forEach(btn => {
            btn.addEventListener("click", (e) => {
                const workoutName = btn.getAttribute("data-workout");
                openBuyWorkoutModal(workoutName);
            });
        });
    }

    // Рендер всей базы клиентов (5 стандартных + текущий пользователь)
    function renderClientsList() {
        const container = document.getElementById("clientsPanel");
        const members = clients.filter(c => c.isMember);
        const nonMembers = clients.filter(c => !c.isMember);
        const membersHtml = `
            <div class="client-list">
                <h3><i class="fas fa-shield-alt"></i> Члены клуба (абонемент есть) — ${members.length}</h3>
                ${members.map(c => `
                    <div class="client-entry">
                        <div class="client-name"><i class="fas fa-user"></i> ${c.name} ${c.id === currentUserId ? '(Ты)' : ''}</div>
                        <div class="trainings-bought">🥊 Тренировки: ${c.purchasedWorkouts.length ? c.purchasedWorkouts.join(", ") : "—"}</div>
                    </div>
                `).join("")}
            </div>
        `;
        const nonMembersHtml = `
            <div class="client-list">
                <h3><i class="fas fa-hourglass-half"></i> Без абонемента — ${nonMembers.length}</h3>
                ${nonMembers.map(c => `
                    <div class="client-entry">
                        <div class="client-name"><i class="fas fa-user-slash"></i> ${c.name} ${c.id === currentUserId ? '(Ты)' : ''}</div>
                        <button class="btn-membership" data-id="${c.id}">Купить абонемент</button>
                        <div class="trainings-bought">📋 тренировок: ${c.purchasedWorkouts.length}</div>
                    </div>
                `).join("")}
            </div>
        `;
        container.innerHTML = membersHtml + nonMembersHtml;
        document.querySelectorAll(".btn-membership").forEach(btn => {
            btn.addEventListener("click", (e) => {
                const id = parseInt(btn.getAttribute("data-id"));
                const client = clients.find(c => c.id === id);
                if (client && !client.isMember) {
                    client.isMember = true;
                    saveToLocal();
                    renderClientsList();
                    updateMyProfileUI();
                    alert(`${client.name} теперь с абонементом!`);
                } else {
                    alert("Уже есть абонемент.");
                }
            });
        });
    }

    function resetDatabase() {
        clients = getDefaultFive();
        // удаляем данные текущего пользователя
        currentUserId = null;
        saveToLocal();
        renderClientsList();
        updateMyProfileUI();
        alert("База сброшена до 5 стандартных клиентов. Твой профиль удалён, создай заново.");
    }

    // Инициализация
    loadFromLocal();
    // Проверим, нет ли сохранённого currentUserId в localStorage? добавим
    const savedUserId = localStorage.getItem("fightClub_currentUserId");
    if (savedUserId) {
        const uid = parseInt(savedUserId);
        if (clients.some(c => c.id === uid)) {
            currentUserId = uid;
        }
    }
    renderTariffs();
    renderWorkouts();
    renderClientsList();
    updateMyProfileUI();

    // События
    document.getElementById("saveMyProfileBtn").onclick = () => {
        const name = document.getElementById("myNameInput").value;
        createOrUpdateMyProfile(name);
        localStorage.setItem("fightClub_currentUserId", currentUserId);
    };
    document.getElementById("confirmPurchaseBtn").onclick = confirmPurchase;
    document.getElementById("closeModalBtn").onclick = closeModal;
    document.getElementById("resetDatabase").onclick = resetDatabase;
